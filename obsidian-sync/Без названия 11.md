## 🔧 **Интеграция с существующей системой:**

Если у вас уже есть когнитивные агенты, добавьте менеджер памяти в их базовый класс:

python

class BaseCognitiveAgent:
    def __init__(self, name):
        self.name = name
        self.memory_manager = memory_manager
    
    def pre_process(self):
        self.memory_block = self.memory_manager.allocate_memory(
            self.name, self.estimated_memory_need()
        )
    
    def post_process(self):
        if hasattr(self, 'memory_block'):
            self.memory_manager.deallocate_memory(self.memory_block)
