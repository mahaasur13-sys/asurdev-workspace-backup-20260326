<script>
// Данные о городах
const cityData = {
  lakhnau: { distance: 48, direction: 'east' },
  delhi: { distance: 15, direction: 'east' },
  mumbai: { distance: 72, direction: 'east' },
  kolkata: { distance: 60, direction: 'east' },
  chennai: { distance: 85, direction: 'east' }
};

// Переключение видимости калькулятора
document.getElementById('varparvritti-toggle').addEventListener('click', function() {
  const container = document.getElementById('varparvritti-container');
  if (container.style.display === 'none') {
    container.style.display = 'block';
    this.textContent = 'Скрыть калькулятор';
  } else {
    container.style.display = 'none';
    this.textContent = 'Показать калькулятор';
  }
});

// Переключение между способами ввода
document.querySelectorAll('input[name="method-obsidian"]').forEach(radio => {
  radio.addEventListener('change', function() {
    if (this.value === 'city') {
      document.getElementById('city-input-obsidian').style.display = 'block';
      document.getElementById('manual-input-obsidian').style.display = 'none';
    } else {
      document.getElementById('city-input-obsidian').style.display = 'none';
      document.getElementById('manual-input-obsidian').style.display = 'block';
    }
  });
});

// Обработчик кнопки расчета
document.getElementById('calculate-btn-obsidian').addEventListener('click', function() {
  let distance, direction;
  
  // Получение данных в зависимости от выбранного метода
  if (document.getElementById('city-method-obsidian').checked) {
    const selectedCity = document.getElementById('city-select-obsidian').value;
    if (!selectedCity) {
      alert('Пожалуйста, выберите город');
      return;
    }
    distance = cityData[selectedCity].distance;
    direction = cityData[selectedCity].direction;
  } else {
    distance = parseFloat(document.getElementById('distance-obsidian').value);
    if (isNaN(distance) || distance <= 0) {
      alert('Пожалуйста, введите корректное расстояние');
      return;
    }
    direction = document.querySelector('input[name="direction-obsidian"]:checked').value;
  }
  
  // Расчет Варправритти
  const result = calculateVarparvritti(distance, direction);
  
  // Отображение результата
  document.getElementById('result-obsidian').style.display = 'block';
  document.getElementById('result-text-obsidian').innerHTML = result.result;
  document.getElementById('calculation-steps-obsidian').textContent = result.steps;
});

// Функция расчета Варправритти
function calculateVarparvritti(distance, direction) {
  let steps = '';
  let result = '';
  
  steps += `Расстояние от Курукшетры: ${distance} йоджан ${direction === 'east' ? 'восточнее' : 'западнее'}\n\n`;
  
  // 1. Отнять 1/4 йоджаны
  const quarterYojana = distance / 4;
  const adjustedDistance = distance - quarterYojana;
  steps += `1. Отнимаем 1/4 йоджаны: ${distance} - ${quarterYojana} = ${adjustedDistance} пал\n`;
  
  // 2. Применить поправку в зависимости от направления
  let timeAdjustment;
  if (direction === 'west') {
    timeAdjustment = adjustedDistance + 15;
    steps += `2. Для западных регионов прибавляем 15 данд: ${adjustedDistance} + 15 = ${timeAdjustment} данд\n`;
  } else {
    timeAdjustment = adjustedDistance - 15;
    steps += `2. Для восточных регионов вычитаем 15 данд: ${adjustedDistance} - 15 = ${timeAdjustment} данд\n`;
  }
  
  // 3. Разница с половиной дня (17.2 данды)
  const halfDay = 17.2;
  const timeDifference = Math.abs(timeAdjustment - halfDay);
  steps += `3. Разница с половиной дня (${halfDay} данд): |${timeAdjustment} - ${halfDay}| = ${timeDifference.toFixed(1)} данд\n`;
  
  // 4. Преобразование в данды и палы
  const fullDandas = Math.floor(timeDifference);
  const remainingPalas = Math.round((timeDifference - fullDandas) * 60);
  steps += `4. Преобразуем в данды и палы: ${timeDifference.toFixed(1)} данд = ${fullDandas} данд ${remainingPalas} пал\n`;
  
  // Формирование итогового результата
  if (direction === 'east') {
    result = `<p><strong>Варправритти в выбранном регионе:</strong> через ${fullDandas} данд ${remainingPalas} пал после восхода солнца</p>`;
  } else {
    result = `<p><strong>Варправритти в выбранном регионе:</strong> за ${fullDandas} данд ${remainingPalas} пал до восхода солнца</p>`;
  }
  
  return {
    result: result,
    steps: steps
  };
}
</script>
