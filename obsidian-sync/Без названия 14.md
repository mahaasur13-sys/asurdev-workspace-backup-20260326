# Калькулятор Варправритти

```dataviewjs
// Конфигурация городов
const cities = {
    "Лакхнау": { distance: 48, direction: "east" },
    "Дели": { distance: 15, direction: "east" },
    "Мумбаи": { distance: 72, direction: "east" },
    "Колката": { distance: 60, direction: "east" },
    "Ченнаи": { distance: 85, direction: "east" }
};

// Функция расчета
function calculateVarparvritti(distance, direction) {
    const quarterYojana = distance / 4;
    const adjustedDistance = distance - quarterYojana;
    
    let timeAdjustment;
    if (direction === "west") {
        timeAdjustment = adjustedDistance + 15;
    } else {
        timeAdjustment = adjustedDistance - 15;
    }
    
    const halfDay = 17.2;
    const timeDifference = Math.abs(timeAdjustment - halfDay);
    const fullDandas = Math.floor(timeDifference);
    const remainingPalas = Math.round((timeDifference - fullDandas) * 60);
    
    return { dandas: fullDandas, palas: remainingPalas, direction: direction };
}

// Создаем интерфейс
const container = this.container;

// Заголовок
container.createEl("h2", { text: "Расчет Варправритти для городов" });

// Таблица результатов
const table = container.createEl("table");
const header = table.createEl("tr");
header.createEl("th", { text: "Город" });
header.createEl("th", { text: "Расстояние" });
header.createEl("th", { text: "Направление" });
header.createEl("th", { text: "Варправритти" });

// Заполняем таблицу
for (const [city, data] of Object.entries(cities)) {
    const result = calculateVarparvritti(data.distance, data.direction);
    const row = table.createEl("tr");
    row.createEl("td", { text: city });
    row.createEl("td", { text: data.distance + " йоджан" });
    row.createEl("td", { text: data.direction === "east" ? "Восточнее" : "Западнее" });
    
    let resultText;
    if (result.direction === "east") {
        resultText = `Через ${result.dandas} данд ${result.palas} пал после восхода`;
    } else {
        resultText = `За ${result.dandas} данд ${result.palas} пал до восхода`;
    }
    
    row.createEl("td", { text: resultText });
}

// Форма для ручного расчета
container.createEl("h3", { text: "Ручной расчет" });

const form = container.createEl("div");
const distanceLabel = form.createEl("div");
distanceLabel.createEl("strong", { text: "Расстояние от Курукшетры (йоджаны):" });
const distanceInput = distanceLabel.createEl("input", { type: "text" });
distanceInput.style.width = "100px";
distanceInput.style.marginLeft = "10px";

const directionLabel = form.createEl("div");
directionLabel.createEl("strong", { text: "Направление:" });
const eastRadio = directionLabel.createEl("input", { type: "radio", name: "direction", value: "east", checked: true });
directionLabel.createEl("span", { text: " Восточнее" });
const westRadio = directionLabel.createEl("input", { type: "radio", name: "direction", value: "west" });
directionLabel.createEl("span", { text: " Западнее" });

const calculateButton = form.createEl("button", { text: "Рассчитать" });
const resultDiv = form.createEl("div");

calculateButton.addEventListener("click", () => {
    const distance = parseFloat(distanceInput.value);
    const direction = eastRadio.checked ? "east" : "west";
    
    if (isNaN(distance) || distance <= 0) {
        resultDiv.textContent = "Пожалуйста, введите корректное расстояние";
        return;
    }
    
    const result = calculateVarparvritti(distance, direction);
    
    let resultText;
    if (result.direction === "east") {
        resultText = `Варправритти: через ${result.dandas} данд ${result.palas} пал после восхода солнца`;
    } else {
        resultText = `Варправритти: за ${result.dandas} данд ${result.palas} пал до восхода солнца`;
    }
    
    resultDiv.textContent = resultText;
});
```
